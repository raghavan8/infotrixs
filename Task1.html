4. Update Package Manager

sudo apt-get update

5. Install Apache2

sudo apt-get install apache2

6. Install PHP and MySQL Support

sudo apt-get install php libapache2-mod-php php-mysql

7. Install MySQL Server

sudo apt-get install mysql-server

8. Connect to MySQL Server

sudo mysql -u root

9. Set MySQL Root User Password

ALTER USER 'root'@'localhost' IDENTIFIED WITH mysql_native_password BY 'Testpassword@123';

10. Create WordPress Database User

CREATE USER 'wp_user'@'localhost' IDENTIFIED BY 'Testpassword@123';

11. Create WordPress Database

CREATE DATABASE wp;

12. Grant Permissions to the WordPress Database

GRANT ALL PRIVILEGES ON wp.* TO 'wp_user'@'localhost';

13. Exit MySQL

exit

14. Download WordPress

cd /tmp wget https://wordpress.org/latest.tar.gz tar -xzvf latest.tar.gz

15. Move WordPress to Apache Document Root

sudo mv wordpress/ /var/www/html

16. Change Directory to WordPress

cd /var/www/html

17. List Contents

ls

18. Restart Apache2

sudo systemctl restart apache2
